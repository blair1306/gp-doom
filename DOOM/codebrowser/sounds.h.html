27//
28// SoundFX struct.
29//
30typedef struct sfxinfo_struct sfxinfo_t;
31
32struct sfxinfo_struct
33{
34 // up to 6-character name
35 char* name;
36
37 // Sfx singularity (only one at a time)
38 int singularity;
39
40 // Sfx priority
41 int priority;
42
43 // referenced sound if a link
44 sfxinfo_t* link;
45
46 // pitch if a link
47 int pitch;
48
49 // volume if a link
50 int volume;
51
52 // sound data
53 void* data;
54
55 // this is checked every second to see if sound
56 // can be thrown out (if 0, then decrement, if -1,
57 // then throw out, if > 0, then it is in use)
58 int usefulness;
59
60 // lump number of sfx
61 int lumpnum;
62};
