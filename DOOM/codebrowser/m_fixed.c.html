54// FixedDiv, C version.
55//
56
57fixed_t
58FixedDiv
59( fixed_t a,
60 fixed_t b )
61{
62 if ( (abs(a)>>14) >= abs(b))
63 return (a^b)<0 ? MININT : MAXINT;
64 return FixedDiv2 (a,b);
65}
