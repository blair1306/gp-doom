507//
508// M_ReadSaveStrings
509// read the strings from the savegame files
510//
511void M_ReadSaveStrings(void)
512{
513 int handle;
514 int count;
515 int i;
516 char name[256];
517
518 for (i = 0;i < load_end;i++)
519 {
520 if (M_CheckParm("-cdrom"))
521 sprintf(name,"c:\\doomdata\\"SAVEGAMENAME"%d.dsg",i);
522 else
523 sprintf(name,SAVEGAMENAME"%d.dsg",i);
524
525 handle = open (name, O_RDONLY | 0, 0666);
526 if (handle == -1)
527 {
528 strcpy(&savegamestrings[i][0],EMPTYSTRING);
529 LoadMenu[i].status = 0;
530 continue;
531 }
532 count = read (handle, &savegamestrings[i], SAVESTRINGSIZE);
533 close (handle);
534 LoadMenu[i].status = 1;
535 }
536}

1293//
1294// Write a string using the hu_font
1295//
1296void
1297M_WriteText
1298( int x,
1299 int y,
1300 char* string)
1301{
1302 int w;
1303 char* ch;
1304 int c;
1305 int cx;
1306 int cy;
1307

1309 ch = string;
1310 cx = x;
1311 cy = y;
1312
1313 while(1)
1314 {

1317 break;
1318 if (c == '\n')
1319 {

1321 cy += 12;
1322 continue;
1323 }
1324
1325 c = toupper(c) - HU_FONTSTART;

1328 cx += 4;

1331
1332 w = SHORT (hu_font[c]->width);
1333 if (cx+w > SCREENWIDTH)
1334 break;

1336 cx+=w;

1400 {

1417 {
1418 ch = KEY_LEFTARROW;

1420 mousex = lastx -= 30;

1423 {
1424 ch = KEY_RIGHTARROW;
1425 mousewait = I_GetTime() + 5;
1426 mousex = lastx += 30;
1427 }

1429 if (ev->data1&1)
1430 {
1431 ch = KEY_ENTER;
1432 mousewait = I_GetTime() + 15;
1433 }
1434
1435 if (ev->data1&2)
1436 {
1437 ch = KEY_BACKSPACE;
1438 mousewait = I_GetTime() + 15;
1439 }
1440 }
1441 else
1442 if (ev->type == ev_keydown)
1443 {
1444 ch = ev->data1;

1448 if (ch == -1)

1453 if (saveStringEnter)
1454 {

1477 ch = toupper(ch);
1478 if (ch != 32)
1479 if (ch-HU_FONTSTART < 0 || ch-HU_FONTSTART >= HU_FONTSIZE)
1480 break;
1481 if (ch >= 32 && ch <= 127 &&
1482 saveCharIndex < SAVESTRINGSIZE-1 &&
1483 M_StringWidth(savegamestrings[saveSlot]) <
1484 (SAVESTRINGSIZE-2)*8)
1485 {
1486 savegamestrings[saveSlot][saveCharIndex++] = ch;
1487 savegamestrings[saveSlot][saveCharIndex] = 0;
1488 }
1489 break;
1490 }
1491 return true;
1492 }

1495 if (messageToPrint)
1496 {
1497 if (messageNeedsInput == true &&

1499 return false;
