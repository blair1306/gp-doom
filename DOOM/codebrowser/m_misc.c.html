62//
63// M_DrawText
64// Returns the final X coordinate
65// HU_Init must have been called to init the font
66//
67extern patch_t* hu_font[HU_FONTSIZE];
68
69int
70M_DrawText
71( int x,
72 int y,
73 boolean direct,
74 char* string )
75{
76 int c;
77 int w;
78
79 while (*string)
80 {
81 c = toupper(*string) - HU_FONTSTART;
82 string++;
83 if (c < 0 || c> HU_FONTSIZE)
84 {
85 x += 4;
86 continue;
87 }
88
89 w = SHORT (hu_font[c]->width);
90 if (x+w > SCREENWIDTH)
91 break;
92 if (direct)
93 V_DrawPatchDirect(x, y, 0, hu_font[c]);
94 else
95 V_DrawPatch(x, y, 0, hu_font[c]);
96 x+=w;
97 }
98
99 return x;
100}

112boolean
113M_WriteFile
114( char const* name,
115 void* source,
116 int length )
117{
118 int handle;
119 int count;
120
121 handle = open ( name, O_WRONLY | O_CREAT | O_TRUNC | O_BINARY, 0666);
122
123 if (handle == -1)
124 return false;
125
126 count = write (handle, source, length);
127 close (handle);
128
129 if (count < length)
130 return false;
131
132 return true;
133}

136//
137// M_ReadFile
138//
139int
140M_ReadFile
141( char const* name,
142 byte** buffer )
143{
144 int handle, count, length;
145 struct stat fileinfo;
146 byte *buf;
147
148 handle = open (name, O_RDONLY | O_BINARY, 0666);
149 if (handle == -1)
150 I_Error ("Couldn't read file %s", name);
151 if (fstat (handle,&fileinfo) == -1)
152 I_Error ("Couldn't read file %s", name);
153 length = fileinfo.st_size;
154 buf = Z_Malloc (length, PU_STATIC, NULL);
155 count = read (handle, buf, length);
156 close (handle);
157
158 if (count < length)
159 I_Error ("Couldn't read file %s", name);
160
161 *buffer = buf;
162 return length;
163}

308//
309// M_SaveDefaults
310//
311void M_SaveDefaults (void)
312{
313 int i;
314 int v;
315 FILE* f;
316
317 f = fopen (defaultfile, "w");
318 if (!f)
319 return; // can't write the file, but don't complain
320
321 for (i=0 ; i<numdefaults ; i++)
322 {
323 if (defaults[i].defaultvalue > -0xfff
324 && defaults[i].defaultvalue < 0xfff)
325 {
326 v = *defaults[i].location;
327 fprintf (f,"%s\t\t%i\n",defaults[i].name,v);
328 } else {
329 fprintf (f,"%s\t\t\"%s\"\n",defaults[i].name,
330 * (char **) (defaults[i].location));
331 }
332 }
333
334 fclose (f);
335}

338//
339// M_LoadDefaults
340//
341extern byte scantokey[128];
342
343void M_LoadDefaults (void)
344{
345 int i;
346 int len;
347 FILE* f;
348 char def[80];
349 char strparm[100];
350 char* newstring;
351 int parm;
352 boolean isstring;
353
354 // set everything to base values
355 numdefaults = sizeof(defaults)/sizeof(defaults[0]);
356 for (i=0 ; i<numdefaults ; i++)
357 *defaults[i].location = defaults[i].defaultvalue;
358
359 // check for a custom default file
360 i = M_CheckParm ("-config");
361 if (i && i<myargc-1)
362 {
363 defaultfile = myargv[i+1];
364 printf (" default file: %s\n",defaultfile);
365 }
366 else
367 defaultfile = basedefault;
368
369 // read the file in, overriding any set defaults
370 f = fopen (defaultfile, "r");
371 if (f)
372 {
373 while (!feof(f))
374 {
375 isstring = false;
376 if (fscanf (f, "%79s %[^\n]\n", def, strparm) == 2)
377 {
378 if (strparm[0] == '"')
379 {
380 // get a string default
381 isstring = true;
382 len = strlen(strparm);
383 newstring = (char *) malloc(len);
384 strparm[len-1] = 0;
385 strcpy(newstring, strparm+1);
386 }
387 else if (strparm[0] == '0' && strparm[1] == 'x')
388 sscanf(strparm+2, "%x", &parm);
389 else
390 sscanf(strparm, "%i", &parm);
391 for (i=0 ; i<numdefaults ; i++)
392 if (!strcmp(def, defaults[i].name))
393 {
394 if (!isstring)
395 *defaults[i].location = parm;
396 else
397 *defaults[i].location =
398 (int) newstring;
399 break;
400 }
401 }
402 }
403
404 fclose (f);
405 }
406}

441//
442// WritePCXfile
443//
444void
445WritePCXfile
446( char* filename,
447 byte* data,
448 int width,
449 int height,
450 byte* palette )
451{
452 int i;
453 int length;
454 pcx_t* pcx;
455 byte* pack;
456
457 pcx = Z_Malloc (width*height*2+1000, PU_STATIC, NULL);
458
459 pcx->manufacturer = 0x0a; // PCX id
460 pcx->version = 5; // 256 color
461 pcx->encoding = 1; // uncompressed
462 pcx->bits_per_pixel = 8; // 256 color
463 pcx->xmin = 0;
464 pcx->ymin = 0;
465 pcx->xmax = SHORT(width-1);
466 pcx->ymax = SHORT(height-1);
467 pcx->hres = SHORT(width);
468 pcx->vres = SHORT(height);
469 memset (pcx->palette,0,sizeof(pcx->palette));
470 pcx->color_planes = 1; // chunky image
471 pcx->bytes_per_line = SHORT(width);
472 pcx->palette_type = SHORT(2); // not a grey scale
473 memset (pcx->filler,0,sizeof(pcx->filler));
474
475
476 // pack the image
477 pack = &pcx->data;
478
479 for (i=0 ; i<width*height ; i++)
480 {
481 if ( (*data & 0xc0) != 0xc0)
482 *pack++ = *data++;
483 else
484 {
485 *pack++ = 0xc1;
486 *pack++ = *data++;
487 }
488 }
489
490 // write the palette
491 *pack++ = 0x0c; // palette ID byte
492 for (i=0 ; i<768 ; i++)
493 *pack++ = *palette++;
494
495 // write output file
496 length = pack - (byte *)pcx;
497 M_WriteFile (filename, pcx, length);
498
499 Z_Free (pcx);
500}

503//
504// M_ScreenShot
505//
506void M_ScreenShot (void)
507{
508 int i;
509 byte* linear;
510 char lbmname[12];
511
512 // munge planar buffer to linear
513 linear = screens[2];
514 I_ReadScreen (linear);
515
516 // find a file name to save it to
517 strcpy(lbmname,"DOOM00.pcx");
518
519 for (i=0 ; i<=99 ; i++)
520 {
521 lbmname[4] = i/10 + '0';
522 lbmname[5] = i%10 + '0';
523 if (access(lbmname,0) == -1)
524 break; // file doesn't exist
525 }
526 if (i==100)
527 I_Error ("M_ScreenShot: Couldn't create a PCX");
528
529 // save the pcx file
530 WritePCXfile (lbmname, linear,
531 SCREENWIDTH, SCREENHEIGHT,
532 W_CacheLumpName ("PLAYPAL",PU_CACHE));
533
534 players[consoleplayer].message = "screen shot";
535}
